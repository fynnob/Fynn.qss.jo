[Unit]
Description=RFCOMM Bluetooth Serial Server
After=bluetooth.service
Requires=bluetooth.service

[Service]
# Restart the listener automatically if it crashes or you disconnect
ExecStart=/bin/bash -c "while true; do /usr/bin/rfcomm watch /dev/rfcomm0 1 /sbin/agetty -L rfcomm0 115200 linux; done"
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
